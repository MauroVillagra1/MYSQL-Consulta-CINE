SELECT peli.CODPELI, peli.TITULO, peli.ANOPROD, gen.GENERO, est.FECESTRENO, est.ESPECTADORES,
ROUND(est.RECAUDACION, 2) AS RECAUDACION, ciu.CIUDAD, pa.PAIS
FROM cine.pelicula peli
INNER JOIN cine.estrenos est ON peli.CODPELI = est.CODPELI
INNER JOIN cine.ciudad ciu ON est.CODCIUDAD = ciu.CODCIUDAD
INNER JOIN cine.pais pa ON ciu.CODPAIS = pa.CODPAIS
INNER JOIN cine.genero gen ON peli.CODGEN = gen.CODGEN
WHERE peli.ANOPROD > 2001
  AND MONTH(est.FECESTRENO) IN (3, 10, 12)
  AND est.RECAUDACION > 20000000
  AND est.ESPECTADORES > 8000000
ORDER BY peli.CODPELI;
